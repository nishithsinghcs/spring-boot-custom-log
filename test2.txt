import cv2
import numpy as np

image_path = '/mnt/data/p3.jpg'
image = cv2.imread(image_path)

# Applying sharpening kernel
kernel = np.array([[0, -1, 0],
                   [-1, 5,-1],
                   [0, -1, 0]])
sharpened = cv2.filter2D(image, -1, kernel)

cv2.imwrite('/mnt/data/sharpened_p3.jpg', sharpened)
